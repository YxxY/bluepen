var kthGrammar = function(N, K) {
    const Kth = (n,k) => {
       
        if(n==1 || k==1) {
            return 0;
        }
        let noOfBits = 2**(n-1);
        let mid = noOfBits/2;
        if(k<=mid) {
            return Kth(n-1,k);
        } else {
            return !Kth(n-1,k-mid);
        }
    }
    
    return Kth(N,K);
    
};
